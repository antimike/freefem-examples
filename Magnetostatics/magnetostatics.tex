\begin{proposition}[Variational form]
				The variational form of the magnetostatic problem is
				\[
					\int_{ \Omega }^{  }  \dfrac{1}{\mu} |\curl{\vb{A}}|^2 \vb{A} - \int_{ \Omega }^{  } (\vb{J} \cdot \vb{A}) \vb{A} = \vb{0} 
				\] 
\end{proposition}
